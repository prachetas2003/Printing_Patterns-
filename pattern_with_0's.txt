#include<iostream>

using namespace std;



int main() {
    int n = 5;
    int i = 1;

    while (i <= n) {
        int j = 1;

        while (j <= i) {
            if (j == 1 || j == i || j == n)
                cout << i << " ";
            else
                cout << "0 ";

            j++;
        }

        cout << endl;
        i++;
    }

    return 0;
}
